#### 建表语句
###### postgresql
```postgresql

  drop table if exists mast_apptree;
  create table mast_apptree (
     mtr_id int8,
     mtr_cobjectid varchar(200) not null,
     mtr_csiteid varchar(30) default(null),
     mtr_cappid varchar(30) default(null),
     mtr_csubappid varchar(30) default(null),
     mtr_cscreenid varchar(30) default(null),
     mtr_cbuttonid varchar(30) default(null),
     mtr_cobjname varchar(100) default(null),
     mtr_cobjnameja varchar(100) default(null),
     mtr_cobjnameen varchar(100) default(null),
     mtr_cobjnamech varchar(100) default(null),
     mtr_cobjname01 varchar(100) default(null),
     mtr_cobjname02 varchar(100) default(null),
     mtr_ctype varchar(1) not null,
     mtr_ctemplateid varchar(30) default(null),
     mtr_curl varchar(300) default(null),
     mtr_cimageurl varchar(300) default(null),
     mtr_csitecaption varchar(100) default(null),
     mtr_csitecaptionja varchar(100) default(null),
     mtr_csitecaptionen varchar(100) default(null),
     mtr_csitecaptionch varchar(100) default(null),
     mtr_csitecaption01 varchar(100) default(null),
     mtr_csitecaption02 varchar(100) default(null),
     mtr_nseq int2 not null,
     mtr_cversion varchar(1) default(null),
     mtr_csystemid varchar(10) default(null),
     mtr_cdefaulttargetuser varchar(1) default(null),
     mtr_ccriterialdatetype varchar(1) default(null),
     mtr_cdatapermissiontype varchar(1) default(null),
     mtr_cappautoload varchar(1) default(null),
     mtr_conlinehelpurl varchar(300) default(null),
     mtr_conlinehelpattr varchar(300) default(null),
     mtr_cdomainid varchar(10) default(null),
     mtr_ciframeflag varchar(1) default(null),
     mtr_cmodifieruserid varchar(30) default(null),
     mtr_dmodifieddate date default(null),
     versionno int8 default(1) not null,
     primary key(mtr_id)
  );
```
